  235.     public function find_all($params = array()) {
  236.         $params = array_merge($this->params, $params);
  237.         
  238.         // Reseta a lista de where list
  239.         $this->where_list = array();
  240.         $this->where_list[] = $params['where'];
  241.         
  242.         
  243.         if(!$this->add_behaviors_content(&$params)){
  244.             return false;
  245.         }
  246.         
  247.         $wheres = '('.join(' AND ',$this->where_list).')';
  248.         
  249.         $sql = "SELECT ".$params['fields']." FROM ".$this->table;
  250.         $sql .= " ".$params['join']." ";
  251.         $sql .=    " WHERE ".$this->add_core_wheres($wheres);
  252.         $sql .= " ".$params['group_by'];
  253.         $sql .= " ORDER BY ".$params['order'];
  254.         $sql .= (!empty($params['limit'])? " LIMIT ".$params['limit']:"");
  255.         
  256.         
  257.         $records = $this->connection->find_with_key($sql,$this->primary_key);
  258.         
  259.         $record_size = count($records);
  260.         if($record_size === 0){
  261.             return false;
  262.         }
  263.         
  264.         
  265.         $records = $this->addCustomFields($records, &$params);
  266.         
  267.         
  268.         //Se não tiver algum include já retorna.
  269.         if(count($params['include']) === 0){
  270.             return $records;
  271.         }
  272.  
  273.         //Adiciona novos atributos do cms ao registro.
  274.         $ids = array();
  275.         foreach ($records as $record) {
  276.             $ids[] = "'".$record['id']."'";
  277.         }
  278.         $ids = join(',',$ids);
  279.         
  280.         // Chama os métodos que incluem os dados extras.
  281.         foreach ($params['include'] as $include) {
  282.             call_user_func_array(array('DrumonModel', 'include_'.$include),array(&$records,$ids));
  283.         }
  284.         
  285.         return $records;
  286.     }
