  107.     public function get_route($route, $app_root) {

  110.         $subfolder = str_replace($_SERVER['DOCUMENT_ROOT'], '', str_replace('\\','/',$app_root));

  116.         if($uri[0] === '' || $uri[0] === '/index.php') {
  117.             if(isset($route[$this->method]['/'])) return $route[$this->method]['/'];
  118.             if(isset($route['*']['/'])) return $route['*']['/'];
  119.         }else{
  120.             
  121.             // Junta as rotas do método com as rotas que aceitam todos os métodos.
  122.             $route_list = array();

  126.             if(isset($route[$this->method])) {
  127.                 $route_list = array_merge($route_list, $route[$this->method]);
  128.                 //if($route_list != NULL) {
  129.                 //$route_list = array_merge($route_list, $route[$this->method]);
  130.                 //}else{
  131.                 //$route_list = $route[$this->method];
  132.                 //}
  133.             }
  134.             // Se não existe rota já retorna false.
  135.             if(empty($route_list)) return false;
  136.             
  137.             // Retorna rota se ela for estática.
  138.             if(isset($route_list[$uri[0]])) {
  139.                 return $route_list[$uri[0]];
  140.             }else if(isset($route_list[$uri[0].'/'])) {
  141.                 return $route_list[$uri[0].'/'];
  142.             }
  143.             
  144.             $uri[0] = substr($uri[0],1);
  145.             $uri_parts = explode('/', $uri[0]);    
  146.             
  147.             foreach ($route_list as $route => $values) {

  150.  
  151.                 // Pula caso seja o root.
  152.                 if(!$route) continue;
  153.                 
  154.                 // Pula se quantidade de partes for diferente.
  155.                 if(sizeof($uri_parts) !== sizeof($route_parts) ) continue;
  156.                 
  157.                 // Pula se parte estática for diferente.
  158.                 $static = explode(':',$route);

  160.                 
  161.                 // Pega as variaveis dinamicas
  162.                 preg_match_all('@:([\w]+)@', $route, $params_name, PREG_PATTERN_ORDER);
  163.  
  164.                 $params_name = $params_name[0];
  165.                 
  166.                 // Se não tem variável então pula.

  168.                 
  169.                 $route_regex = $route;

  176.                     // TODO: rever se vai ficar ou não o preg no lugar do str

  181.                 if(preg_match('/'.$route_regex.'/' , $uri[0], $matches) === 1){

  205.     public function url_for($named_route,$params = array()) {
  206.

  215.         $path = false;
  216.         foreach ($route_list as $url => $route) {

  218.                 
  219.                 // Pega as variaveis dinamicas
  220.                 preg_match_all('@:([\w]+)@', $url, $params_name, PREG_PATTERN_ORDER);
  221.  
  222.                 $params_name = $params_name[0];
  223.                 
  224.                 // Se não tem variável então pula.
  225.                 if(count($params_name)){
  226.                     if (count($params) != count($params_name)) { die('Named route for '.$named_route.' expects '.count($params_name).' params not '.count($params).'.'); }
  227.                     $route_regex = $url;

  229.                     
  230.                     $i = 0;
  231.                     foreach ($params_name as $name) {
  232.                         $route_regex = preg_replace('/'.$name.'/',$params[$i],$route_regex,1);
  233.                         $i++;

  236.                 }else{
  237.                     $path = $url;
  238.                 }
  239.             }
  240.         }
  241.         
  242.         if(!$path) die('Named route for '.$named_route.' doenst exist.');
  243.         
  244.         return str_replace('\/','/',$path); ;
  245.     }
