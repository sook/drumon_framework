  369. function findSharp($orig, $final) // function from Ryan Rud (http://adryrun.com)
  370. {
  371.     $final    = $final * (750.0 / $orig);
  372.     $a        = 52;
  373.     $b        = -0.27810650887573124;
  374.     $c        = .00047337278106508946;
  375.     
  376.     $result = $a + $b * $final + $c * $final * $final;
  377.     
  378.     return max(round($result), 0);
  379. } // findSharp()

  381. function doConditionalGet($etag, $lastModified)
  382. {
  383.     header("Last-Modified: $lastModified");
  384.     header("ETag: \"{$etag}\"");
  385.         
  386.     $if_none_match = isset($_SERVER['HTTP_IF_NONE_MATCH']) ?
  387.         stripslashes($_SERVER['HTTP_IF_NONE_MATCH']) : 
  388.         false;
  389.     
  390.     $if_modified_since = isset($_SERVER['HTTP_IF_MODIFIED_SINCE']) ?
  391.         stripslashes($_SERVER['HTTP_IF_MODIFIED_SINCE']) :
  392.         false;
  393.     
  394.     if (!$if_modified_since && !$if_none_match)
  395.         return;
  396.     
  397.     if ($if_none_match && $if_none_match != $etag && $if_none_match != '"' . $etag . '"')
  398.         return; // etag is there but doesn't match
  399.     
  400.     if ($if_modified_since && $if_modified_since != $lastModified)
  401.         return; // if-modified-since is there but doesn't match
  402.     
  403.     // Nothing has changed since their last request - serve a 304 and exit
  404.     header('HTTP/1.1 304 Not Modified');
  405.     exit();
  406. } // doConditionalGet()
