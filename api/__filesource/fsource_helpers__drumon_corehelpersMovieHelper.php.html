  31.     private function parse_url($url) {
  32.         if (preg_match('/watch\?v\=([A-Za-z0-9_-]+)/', $url, $matches)) {
  33.             $this->location = 'youtube';
  34.             return $matches[1];
  35.         }
  36.         if (preg_match('/([0-9]+)/', $url, $matches)) {
  37.              $this->location = 'vimeo';

  39.         }
  40.         return false;
  41.     }
